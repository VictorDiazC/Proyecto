a.

Para n = 100, 1000, 10000, 100000 haga n simulaciones para obtener la función de masa de probabilidad aproximada de X.

caras_del_dado <- 1:6
num_dados <- 3
num_sim <- 100000
vector_simuls <- c()
for(i in 1:num_sim){
  simulacion <- sample(x = caras_del_dado, size = num_dados, replace = TRUE)
  valor_simulacion <- sum(simulacion)
  vector_simuls <- c(vector_simuls,valor_simulacion)
}
